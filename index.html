factorial using functions

#include void factorial();
int main()
{
factorial();
return 1;
}
void factorial()
{
int i,n,f=1;
printf("Enter a number: ");
scanf("%d",&n);
for(i=1;i<=n;i++)
{
f=f*i;
}
printf("factorial %d is: %d\n",n,f);
}

prime or not using functions

#include
void prime();
int main()
{
prime();
return 1;
}
void prime()
{
int i,n,count=0;
printf("Enter a number: ");
scanf("%d",&n);
for(i=1;i<=n;i++)
{
if(n%i==0)
count++;
}
if(count==2)
{
printf("%d is a prime number",n);
}
else
{
printf("%d is not a prime number",n);
}
}

void alphabet()
{
char ch;

scanf("%c",&ch);
if((ch>='a'&&ch<='z')||(ch>='A'&&ch<='Z'))
printf("It is an alphabet\n");
else
printf("It is not an alphabet\n");
}

call by value

#include
void modify(int n);
int main()
{
int n=10;
printf("Before modify n= %d\n",n);
modify(n);
printf("After modify n= %d\n",n);
return 1;
}
void modify(int n)
{
n=n+10;
printf("From modify n= %d\n",n);
}

call by reference

#include
void modify(int *p);
int main()
{
int n=10;
printf("Before modify n= %d\n",n);
modify(&n);
printf("After modify n= %d\n",n);
return 1;
}
void modify(int *p)
{
*p=*p+10;
}
